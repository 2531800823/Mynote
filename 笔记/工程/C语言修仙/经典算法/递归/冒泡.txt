#include<stdio.h>
#include<stdlib.h>

int Max(int a[], int L, int R)
{
	if (L < R)
	{
		int i;
		for (i = L; i <= R-1; i++)
		{
			if (a[i] > a[i + 1])
			{
				int tamp = a[i];
				a[i] = a[i + 1];
				a[i + 1] = tamp;
			}
		}
		Max(a, L, R - 1);
	}

}

int main(void)
{
	int a[10] = { 1,2,3,4,10,6,7,8,9,0};
	int i;
	Max(a,0,9);
	for (i = 0; i < 10; i++) {
		printf("%d\n", a[i]);
	}

	system("pause");
	return 0;
