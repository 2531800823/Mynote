#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>


int main()
{
	char ch;
	int num;
	int i, j, k, l;
	scanf("%c",&ch);
	num = ch - 'A';
	for (i = 0; i <= num + 1; i++) {
		for (j = i; j < num; j++)
			printf(" ");
		for (k = 0; k < i + 1; k++)
			printf("%c", 'A' + k);
		for (l = i; l > 0; l--)
			printf("%c", 'A' + (l - 1));
		printf("\n");
	}



	system("pause");
	return 0;
}
